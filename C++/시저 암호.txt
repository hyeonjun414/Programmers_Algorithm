#include <string>
#include <vector>
#include <iostream>
using namespace std;

string solution(string s, int n) {
    string answer = "";
    for(int i=0; i<s.size(); i++)
    {
        if(islower(s[i]))
            answer += char((s[i]-'a'+ n)%26+'a');
        else if(isupper(s[i]))
            answer += char((s[i]-'A'+ n)%26+'A');
        else
            answer += ' ';
    }
    return answer;
}
